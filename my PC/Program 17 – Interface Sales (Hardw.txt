Program 17 – Interface: Sales (HardwareItem & SoftwareItem)
A company sells both hardware and software. Define an interface Sales with variable SalesFigure = 1000 and method displaySalesData(). Create classes HardwareItem and SoftwareItem that extend class Item and implement interface Sales. Each class has its own methods to get data and display data.

Code:

import java.util.Scanner;

interface Sales {
    void salesFigure();
}

class Item {
    String title;
    double price;

    void getItem(java.util.Scanner sc) {
        System.out.println("Enter item title: ");
        title = sc.nextLine();
        System.out.print("Enter price: ");
        price = sc.nextDouble();
        sc.nextLine();
    }

    void showItem() {
        System.out.println("Title: " + title);
        System.out.println("Price: " + price);
    }
}

class HardwareItem extends Item implements Sales {
    int id;
    String manufacturer;

    void getHardware(java.util.Scanner sc) {
        System.out.print("Enter hardware id: ");
        id = sc.nextInt();
        sc.nextLine();
        System.out.print("Enter manufacturer: ");
        manufacturer = sc.nextLine();
    }

    void showHardware() {
        System.out.println("Hardware ID: " + id);
        System.out.println("Manufacturer ID: " + id);
    }

    public void salesFigure() {
        System.out.println("Hardware Sales Calculated");
    }
}

class SoftwareItem extends Item implements Sales {
    int id;
    String operatingSystem;

    void getSoftware(java.util.Scanner sc) {
        System.out.print("Enter software id: ");
        id = sc.nextInt();
        sc.nextLine();
        System.out.print("Enter operating system: ");
        operatingSystem = sc.nextLine();
    }

    void showSoftware() {
        System.out.println("Software ID: " + id);
        System.out.println("Operating system: " + operatingSystem);
    }

    public void salesFigure() {
        System.out.println("Software Sales Calculated");
    }
}

class Sales {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.println("Hardware Item");
        HardwareItem h = new HardwareItem();
        h.getItem(sc);
        h.getHardware(sc);
        h.showItem();
        h.showHardware();
        h.salesFigure();

        System.out.println("\n-- Software Item --");
        SoftwareItem s = new SoftwareItem();
        s.getItem(sc);
        s.getSoftware(sc);
        s.showItem();
        s.showSoftware();
        s.salesFigure();

        sc.close();
    }
}
